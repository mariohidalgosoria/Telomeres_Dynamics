#Delta t=20

Deltat=20;
TR=0;
for i=2:364
 name=sprintf('CCLAT%d.dat',i);
 data=load(name);
 TM=0;
 for i=1:(rows(data)-Deltat)
    DX=data(i+Deltat,2)-data(i,2);
    DY=data(i+Deltat,3)-data(i,3);
    R=sqrt(DX^2 + DY^2);
    TM=[TM;R];
 end 
TM(1,:)=[];
R=mean(TM)/max(TM);
TR=[TR;R];
end
TR(1,:)=[];


PR=0;
for i=2:364
 name=sprintf('MANGD20T%d.dat',i);
 data=load(name);
 P=data(:,2);
 P=P/pi;
 count=0;
 PB=0; 
 for i=1:(rows(data))
   if(P(i)<= 0.1)
     count++;
   end
 end
PB=count/rows(data);
PR=[PR;PB]; 
end

PR(1,:)=[];
plot(PR,TR,'.')

SS=[PR,TR];
save PRWTCDT20.dat SS -ascii
corr(PR,TR)
